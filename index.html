                    struct Estutura_Fila *Cria_Fila(int Numero_Fila){
                    
                    struct Estrutura_Fila * Cria_F;
                    
                    Cria_F=(struct Estrutura_Fila *)malloc(sizeof(struct Estrutura_Fila));
                    Cria_F->Topo=0;
                    Cria_F->Inicio=0;
                    Cria_F->Tamanho=0;
                    Cria_F->Tamanho_Maximo=Numero_Fila;
                    Cria_F->Numero=(int *)malloc(Numero_Fila*sizeof(int));
                    
                    return Cria_F;
                    }
                    
                    void Adciona_Fila(struct Estrutura_Fila *Fila,int Numero){
